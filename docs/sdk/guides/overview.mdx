## Small Patterns

### Streaming CLI with Agent

```typescript
import { Agent } from "@clinebot/agents"

const agent = new Agent({
  providerId: "anthropic",
  modelId: "claude-sonnet-4-6",
  apiKey: process.env.ANTHROPIC_API_KEY,
  systemPrompt: "You are a concise coding assistant.",
  tools: [],
  onEvent: (event) => {
    if (event.type === "content_end" && event.contentType === "text") {
      process.stdout.write(event.text ?? "")
    }
  },
})

await agent.run("Explain this repository structure")
```

### Persistent Session with ClineCore

```typescript
import { ClineCore } from "@clinebot/core"

const cline = await ClineCore.create({ clientName: "example" })

const session = await cline.start({
  prompt: "Summarize this codebase",
  config: {
    providerId: "anthropic",
    modelId: "claude-sonnet-4-6",
    apiKey: process.env.ANTHROPIC_API_KEY,
    systemPrompt: "You are a helpful coding assistant.",
    cwd: process.cwd(),
    enableTools: true,
    enableSpawnAgent: false,
    enableAgentTeams: false,
  },
})

await cline.send({
  sessionId: session.sessionId,
  prompt: "Now identify risky files",
})
```
